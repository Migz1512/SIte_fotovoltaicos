Código
int pinoA0 = 0;
void setup()
{ pinMode(A0, INPUT);
pinMode(11, OUTPUT);
pinMode(13, OUTPUT);
pinMode(12, OUTPUT);
}

void loop()
{
pinoA0 = analogRead(A0);
if (pinoA0 < 300) {
digitalWrite(11, LOW);
digitalWrite(13, LOW);
digitalWrite(12, LOW);
}
if (pinoA0 >= 300 && pinoA0 < 600) {
digitalWrite(13, HIGH);
digitalWrite(12, LOW);
digitalWrite(11, LOW);
}
if (pinoA0 >= 600 && pinoA0 < 900) {
digitalWrite(12, HIGH);
digitalWrite(13, LOW);
digitalWrite(11, LOW);
}
if (pinoA0 >= 900) {
digitalWrite(11, HIGH);
digitalWrite(13, LOW);
digitalWrite(12, LOW);
}
delay(10); // Delay a little bit to improve simulation performance
